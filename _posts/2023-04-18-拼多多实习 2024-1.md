#include <bits/stdc++.h>
using namespace std;


int main() {
    //
    int X, Y;
    cin >> X >> Y;
    if (X < 2 || Y < 2) {
        cout << 0 << endl;
        return 0;
    }
    int ans{};
    for (int k{1}; k < X; k += 2)
        for (int p{1}; p < Y; p += 2) ans += (X - k) * (Y - p);
    cout << ans << endl;
}
